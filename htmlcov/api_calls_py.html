46def search_amazon(query_text): 

47 """ 

48 Performs an Amazon search with given query_text 

49 Returns the List of search results as dicts 

50 Each dict contains "ASIN", "title", "imageUrl", "detailPageURL" and others as keys 

51 Note: 150 calls/mo + 0.01/extra call 

52 """ 

53 headers = { 

54 'x-rapidapi-key': RAPID_API_KEY, 

55 'x-rapidapi-host': "amazon-price1.p.rapidapi.com" 

56 } 

57 params = { 

58 "keywords": query_text, 

59 "marketplace":"US" 

60 } 

61 resp = requests.get(API_URL_SEARCH, headers=headers, params=params) 

62 

63 if resp.status_code != 200: 

64 # No json if error 

65 response_data = "There was an error with getting amazon search results. Error: {}"\ 

66 .format(resp.status_code) 

67 print(response_data) 

68 return response_data 

69 print("Search response OK") 

70 response_data = resp.json() 

71 return response_data
