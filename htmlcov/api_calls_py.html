55def search_amazon(query_text): 

56 """ 

57 Performs an Amazon search with given query_text 

58 Returns the List of search results as dicts 

59 Each dict contains "ASIN", "title", "imageUrl", "detailPageURL" and others as keys 

60 Note: 150 calls/mo + 0.01/extra call 

61 """ 

62 headers = { 

63 'x-rapidapi-key': RAPID_API_KEY, 

64 'x-rapidapi-host': "amazon-price1.p.rapidapi.com" 

65 } 

66 params = { 

67 "keywords": query_text, 

68 "marketplace":"US" 

69 } 

70 resp = requests.get(API_URL_SEARCH, headers=headers, params=params) 

71 if resp.status_code != 200: 

72 print("There was an error with getting amazon search results. Error: {}"\ 

73 .format(resp.status_code)) 

74 return None 

75 print("Search response OK") 

76 response_data = resp.json() 

77 return response_data

79def fetch_price_history(asin): 

80 """ 

81 Takes the ASIN of an amazon product

85 """

91 params = { 

92 "asin": asin,

94 } 

95
